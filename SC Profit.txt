    <script>
        let users = JSON.parse(localStorage.getItem('scProfitUsers')) || [];
        let currentUser = null;

        function openWhatsApp(type = 'support') {
            const phoneNumber = "03043360034";
            let message = "";
            
            if (type === 'withdraw') {
                const userName = currentUser ? currentUser.name : "User";
                const userBalance = currentUser ? currentUser.balance : 0;
                message = `Withdraw Request\nName: ${userName}\nBalance: ‚Ç®${userBalance}\nPlease process my withdrawal.`;
            } else if (type === 'deposit_help') {
                message = "Hello! I need help with deposit on SC Profit.";
            } else {
                message = "Hello! I need support from SC Profit platform.";
            }
            
            window.open(`https://wa.me/${phoneNumber}?text=${encodeURIComponent(message)}`, '_blank');
        }

        function showLoginPage() {
            hideAllPages();
            document.getElementById('loginPage').classList.add('active');
            clearErrors();
        }
        
        function showRegisterPage() {
            hideAllPages();
            document.getElementById('registerPage').classList.add('active');
            clearErrors();
        }
        
        function showDashboardPage() {
            if (!currentUser) {
                showLoginPage();
                return;
            }
            hideAllPages();
            document.getElementById('dashboardPage').classList.add('active');
            updateDashboard();
        }
        
        function showDepositPage() {
            hideAllPages();
            document.getElementById('depositPage').classList.add('active');
        }
        
        function showWithdrawPage() {
            if (!currentUser) {
                showLoginPage();
                return;
            }
            hideAllPages();
            document.getElementById('withdrawPage').classList.add('active');
            updateWithdrawPage();
        }
        
        function hideAllPages() {
            document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
        }
        
        function clearErrors() {
            document.getElementById('loginError').style.display = 'none';
            document.getElementById('registerError').style.display = 'none';
        }
        
        function showError(elementId, message) {
            const errorElement = document.getElementById(elementId);
            errorElement.textContent = message;
            errorElement.style.display = 'block';
        }
        
        function updateWithdrawPage() {
            if (currentUser) {
                document.getElementById('availableBalance').textContent = currentUser.balance;
            }
        }

        function register() {
            const name = document.getElementById('regName').value;
            const email = document.getElementById('regEmail').value;
            const password = document.getElementById('regPassword').value;
            const confirmPassword = document.getElementById('regConfirmPassword').value;
            
            clearErrors();
            
            if (!name || !email || !password || !confirmPassword) {
                showError('registerError', 'Please fill all fields');
                return;
            }
            
            if (password.length < 6) {
                showError('registerError', 'Password must be at least 6 characters');
                return;
            }
            
            if (password !== confirmPassword) {
                showError('registerError', 'Passwords do not match');
                return;
            }
            
            const existingUser = users.find(user => user.email === email);
            if (existingUser) {
                showError('registerError', 'User with this email already exists');
                return;
            }
            
            const newUser = { name: name, email: email, password: password, balance: 0 };
            users.push(newUser);
            localStorage.setItem('scProfitUsers', JSON.stringify(users));
            currentUser = newUser;
            showDashboardPage();
        }
        
        function login() {
            const email = document.getElementById('loginEmail').value;
            const password = document.getElementById('loginPassword').value;
            
            clearErrors();
            
            if (!email || !password) {
                showError('loginError', 'Please enter email and password');
                return;
            }
            
            const user = users.find(u => u.email === email && u.password === password);
            if (user) {
                currentUser = user;
                showDashboardPage();
            } else {
                showError('loginError', 'Invalid email or password');
            }
        }
        
        function updateDashboard() {
            if (currentUser) {
                document.getElementById('userName').textContent = currentUser.name;
                document.getElementById('userBalance').textContent = currentUser.balance;
            }
        }
        
        function processDeposit() {
            const amount = parseFloat(document.getElementById('depositAmount').value);
            
            if (!amount || amount <= 0) {
                alert('Please enter valid amount');
                return;
            }
            
            if (currentUser) {
                currentUser.balance += amount;
                const userIndex = users.findIndex(u => u.email === currentUser.email);
                if (userIndex !== -1) {
                    users[userIndex] = currentUser;
                    localStorage.setItem('scProfitUsers', JSON.stringify(users));
                }
                alert(`Deposit of ‚Ç®${amount} successful!`);
                document.getElementById('depositAmount').value = '';
                showDashboardPage();
            }
        }
        
        function logout() {
            currentUser = null;
            showLoginPage();
            document.getElementById('loginEmail').value = '';
            document.getElementById('loginPassword').value = '';
        }
